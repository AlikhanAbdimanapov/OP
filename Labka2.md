# сложение
a = 7 + 3
print("7 + 3 =", a)

# вычитание
b = 10 - 4
print("10 - 4 =", b)

# умножение
c = 6 * 5
print("6 * 5 =", c)

# возведение в степень
d = 2 ** 3
print("2 ** 3 =", d)

# деление (обычное, результат дробный)
e = 9 / 2
print("9 / 2 =", e)

# целочисленное деление
f = 9 // 2
print("9 // 2 =", f)

# остаток от деления
g = 9 % 2
print("9 % 2 =", g)
